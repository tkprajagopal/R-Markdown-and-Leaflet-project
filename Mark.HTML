map <- mimapa %>%
  leaflet() %>%
  addTiles() %>%
  addMarkers(popup=paste

              "<br>City: ", 
               htmlEscape(mimapa$City), 
              "<br>Restaurant: ", 
               htmlEscape(mimapa$Name),
